Laravel Integration

Service Provider Integration

TaskRunner integrates seamlessly with Laravel's service container:

// Register TaskRunner service provider
// config/app.php
'providers' => [
    // ...
    App\Modules\TaskRunner\TaskRunnerServiceProvider::class,
],

// Publish configuration
php artisan vendor:publish --provider="App\Modules\TaskRunner\TaskRunnerServiceProvider"

// Register facades
'aliases' => [
    // ...
    'TaskRunner' => App\Modules\TaskRunner\Facades\TaskRunner::class,
    'TaskDispatcher' => App\Modules\TaskRunner\Facades\TaskDispatcher::class,
]

Service Container

  • TaskDispatcher: Main task dispatching service
  • TaskService: Task management and CRUD operations
  • AnalyticsService: Performance analytics and metrics
  • MonitoringService: System monitoring and alerting

Facades

  • TaskRunner: Main facade for task operations
  • TaskDispatcher: Task dispatching facade
  • TaskAnalytics: Analytics and metrics facade
  • TaskMonitoring: Monitoring and alerting facade

Artisan Commands

TaskRunner provides Artisan commands for task management:

// List available tasks
php artisan task:list

// Run a specific task
php artisan task:run database-backup

// Run task with options
php artisan task:run deployment --environment=production --version=v2.1.0

// Show task details
php artisan task:show database-backup

// Run task chain
php artisan task:chain deployment-workflow

// Monitor task execution
php artisan task:monitor task_123

// View task analytics
php artisan task:analytics --period=7d

// Clean up old tasks
php artisan task:cleanup --older-than=30d

Eloquent Integration

TaskRunner integrates with Laravel's Eloquent ORM:

// Task model with relationships
class Task extends Model
{
    protected $fillable = [
        'name', 'status', 'script', 'output', 'started_at', 'completed_at'
    ];

    protected $casts = [
        'started_at' => 'datetime',
        'completed_at' => 'datetime',
        'options' => 'array',
        'metadata' => 'array'
    ];

    // Relationships
    public function user()
    {
        return $this->belongsTo(User::class);
    }

    public function chain()
    {
        return $this->belongsTo(TaskChain::class);
    }

    public function analytics()
    {
        return $this->hasMany(TaskAnalytic::class);
    }

    public function monitoringEvents()
    {
        return $this->hasMany(MonitoringEvent::class);
    }
}

// Query tasks with relationships
$tasks = Task::with(['user', 'analytics'])
    ->where('status', 'completed')
    ->where('created_at', '>=', now()->subDays(7))
    ->get();

// Create task with relationships
$task = Task::create([
    'name' => 'database-backup',
    'user_id' => auth()->id(),
    'script' => 'pg_dump production_db > backup.sql'
]);

WebSocket API

Real-time WebSocket API for task monitoring:

// WebSocket connection
// Connect to WebSocket endpoint
const socket = new WebSocket('ws://localhost:6001/task-events');

// Subscribe to task events
socket.send(JSON.stringify({
    action: 'subscribe',
    channel: 'task-events',
    task_id: 'task_123'
}));

// Listen for events
socket.onmessage = function(event) {
    const data = JSON.parse(event.data);
    
    switch(data.event) {
        case 'task.started':
            console.log('Task started:', data.task_id);
            break;
        case 'task.progress':
            updateProgress(data.percentage, data.message);
            break;
        case 'task.completed':
            console.log('Task completed:', data.task_id);
            break;
        case 'task.failed':
            console.log('Task failed:', data.error);
            break;
    }
};

// Server-side WebSocket handling
class TaskWebSocketHandler
{
    public function onMessage($connection, $data)
    {
        $message = json_decode($data, true);
        
        switch($message['action']) {
            case 'subscribe':
                $connection->subscribe($message['channel']);
                break;
            case 'unsubscribe':
                $connection->unsubscribe($message['channel']);
                break;
        }
    }
}

Livewire Components

Task Dashboard Component

Real-time task dashboard with Livewire:

// TaskDashboard Livewire component
class TaskDashboard extends Component
{
    public $tasks = [];
    public $stats = [];
    public $recentActivity = [];

    protected $listeners = [
        'taskStarted' => 'refreshTasks',
        'taskCompleted' => 'refreshTasks',
        'taskFailed' => 'refreshTasks'
    ];

    public function mount()
    {
        $this->loadTasks();
        $this->loadStats();
        $this->loadRecentActivity();
    }

    public function loadTasks()
    {
        $this->tasks = Task::with('user')
            ->latest()
            ->limit(10)
            ->get()
            ->toArray();
    }

    public function loadStats()
    {
        $this->stats = [
            'total_tasks' => Task::count(),
            'running_tasks' => Task::where('status', 'running')->count(),
            'completed_tasks' => Task::where('status', 'completed')->count(),
            'failed_tasks' => Task::where('status', 'failed')->count(),
        ];
    }

    public function runTask($taskId)
    {
        $task = Task::find($taskId);
        TaskDispatcher::dispatch($task);
        
        $this->emit('taskStarted', $taskId);
    }

    public function render()
    {
        return view('livewire.task-dashboard');
    }
}

Task Monitor Component

Real-time task monitoring with progress updates:

// TaskMonitor Livewire component
class TaskMonitor extends Component
{
    public $taskId;
    public $task = null;
    public $output = '';
    public $progress = 0;
    public $status = '';

    protected $listeners = [
        'echo:task-events,TaskProgress' => 'handleProgress',
        'echo:task-events,TaskCompleted' => 'handleCompleted',
        'echo:task-events,TaskFailed' => 'handleFailed'
    ];

    public function mount($taskId)
    {
        $this->taskId = $taskId;
        $this->loadTask();
    }

    public function loadTask()
    {
        $this->task = Task::find($this->taskId);
        $this->status = $this->task->status;
        $this->output = $this->task->output;
    }

    public function handleProgress($event)
    {
        if ($event['task_id'] === $this->taskId) {
            $this->progress = $event['percentage'];
            $this->output = $event['message'];
        }
    }

    public function handleCompleted($event)
    {
        if ($event['task_id'] === $this->taskId) {
            $this->status = 'completed';
            $this->progress = 100;
            $this->loadTask();
        }
    }

    public function handleFailed($event)
    {
        if ($event['task_id'] === $this->taskId) {
            $this->status = 'failed';
            $this->output = $event['error'];
        }
    }

    public function render()
    {
        return view('livewire.task-monitor');
    }
}

Deployment Integration

Complete deployment workflow integration:

// Deployment integration with external systems
class DeploymentIntegration
{
    public function handleDeploymentStarted(DeploymentStartedEvent $event)
    {
        // Notify CI/CD system
        $this->notifyCICD($event);
        
        // Update deployment status
        $this->updateDeploymentStatus($event);
        
        // Send Slack notification
        $this->sendSlackNotification($event);
    }

    public function handleDeploymentCompleted(DeploymentCompletedEvent $event)
    {
        // Update deployment status
        $this->updateDeploymentStatus($event);
        
        // Run post-deployment tests
        $this->runPostDeploymentTests($event);
        
        // Send completion notification
        $this->sendCompletionNotification($event);
    }

    private function notifyCICD(DeploymentStartedEvent $event)
    {
        Http::post(config('deployment.cicd_webhook'), [
            'deployment_id' => $event->deploymentId,
            'environment' => $event->environment,
            'version' => $event->version,
            'status' => 'started',
            'timestamp' => now()->toISOString()
        ]);
    }

    private function updateDeploymentStatus($event)
    {
        // Update deployment tracking system
        Http::put("https://deploy.example.com/api/deployments/{$event->deploymentId}", [
            'status' => $event->status,
            'completed_at' => $event->completedAt ?? null,
            'duration' => $event->duration ?? null
        ]);
    }

    private function sendSlackNotification($event)
    {
        $message = $event->status === 'completed' 
            ? "✅ Deployment completed successfully"
            : "🚀 Deployment started";

        Http::post(config('services.slack.webhook_url'), [
            'text' => $message,
            'attachments' => [
                [
                    'fields' => [
                        ['title' => 'Environment', 'value' => $event->environment, 'short' => true],
                        ['title' => 'Version', 'value' => $event->version, 'short' => true],
                        ['title' => 'Duration', 'value' => $event->duration . 's', 'short' => true]
                    ]
                ]
            ]
        ]);
    }
}

Monitoring Integration

Integration with monitoring and alerting systems:

// Monitoring system integration
class MonitoringIntegration
{
    public function handleTaskFailed(TaskFailedEvent $event)
    {
        // Send to monitoring system
        $this->sendToMonitoring($event);
        
        // Create incident if needed
        $this->createIncident($event);
        
        // Send alert to team
        $this->sendAlert($event);
    }

    public function handlePerformanceAlert(PerformanceAlertEvent $event)
    {
        // Send to performance monitoring
        $this->sendToPerformanceMonitoring($event);
        
        // Scale resources if needed
        $this->scaleResources($event);
    }

    private function sendToMonitoring(TaskFailedEvent $event)
    {
        Http::post(config('monitoring.webhook_url'), [
            'alert_type' => 'task_failure',
            'severity' => 'high',
            'task_id' => $event->taskId,
            'task_name' => $event->taskName,
            'error' => $event->error,
            'failed_at' => $event->failedAt->toISOString(),
            'environment' => config('app.env')
        ]);
    }

    private function createIncident(TaskFailedEvent $event)
    {
        // Create incident in incident management system
        Http::post(config('incident.management_url'), [
            'title' => "Task {$event->taskName} failed",
            'description' => $event->error,
            'severity' => 'high',
            'assignee' => $this->getOnCallEngineer(),
            'tags' => ['taskrunner', 'automation', 'failure']
        ]);
    }

    private function sendAlert(TaskFailedEvent $event)
    {
        // Send to PagerDuty
        Http::post(config('pagerduty.webhook_url'), [
            'routing_key' => config('pagerduty.routing_key'),
            'event_action' => 'trigger',
            'payload' => [
                'summary' => "Task {$event->taskName} failed",
                'severity' => 'critical',
                'source' => 'taskrunner',
                'custom_details' => [
                    'task_id' => $event->taskId,
                    'error' => $event->error,
                    'failed_at' => $event->failedAt->toISOString()
                ]
            ]
        ]);
    }
}
