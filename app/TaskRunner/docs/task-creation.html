Anonymous Tasks

Quick Task Creation

Anonymous tasks are the fastest way to create and execute simple tasks without defining custom classes:

use App\Modules\TaskRunner\AnonymousTask;

// Simple command task
$task = AnonymousTask::command('backup-database', 'mysqldump -u root -p myapp > backup.sql');

// Script task with options
$task = AnonymousTask::script('system-check', 'echo "System check completed"', [
    'timeout' => 60,
    'user' => 'www-data'
]);

// View task with data
$task = AnonymousTask::view('deployment-status', 'tasks.deployment-status', [
    'environment' => 'production',
    'version' => '1.2.3'
]);

// Callback task
$task = AnonymousTask::callback('process-data', function() {
    return 'Data processed successfully';
});

// Inline task
$task = AnonymousTask::inline('quick-task', 'echo "Hello World"');

// Multiple commands
$task = AnonymousTask::commands('setup-environment', [
    'composer install',
    'npm install',
    'php artisan migrate',
    'php artisan config:cache'
]);

// Task with environment variables
$task = AnonymousTask::withEnv('deploy-app', [
    'APP_ENV' => 'production',
    'DB_CONNECTION' => 'mysql'
], 'php artisan deploy');

// Conditional task
$task = AnonymousTask::conditional('conditional-backup', [
    'if' => 'test -f /var/backups/latest.sql',
    'then' => 'echo "Backup exists"',
    'else' => 'mysqldump -u root -p myapp > /var/backups/latest.sql'
]);

// Task with error handling
$task = AnonymousTask::withErrorHandling(
    'safe-deployment',
    'php artisan deploy',
    'php artisan rollback'
);

// Task with retry logic
$task = AnonymousTask::withRetry(
    'api-call',
    'curl -f https://api.example.com/health',
    3, // max retries
    5   // delay between retries
);

// Task with progress tracking
$task = AnonymousTask::withProgress('long-running-task', [
    'step1' => 'echo "Step 1 completed"',
    'step2' => 'echo "Step 2 completed"',
    'step3' => 'echo "Step 3 completed"'
]);

// Task with cleanup
$task = AnonymousTask::withCleanup(
    'temporary-operation',
    'cp important-file.txt /tmp/backup.txt',
    'rm /tmp/backup.txt'
);

// Task with logging
$task = AnonymousTask::withLogging(
    'logged-operation',
    'echo "Operation completed"',
    '/var/log/taskrunner.log'
);

Anonymous Task Options

Option Type Default Description
timeout int 300 Task execution timeout in seconds
user string null User to run the task as
cwd string null Working directory for the task
env array [] Environment variables
background bool false Run task in background
retry int 0 Number of retry attempts
retry_delay int 5 Delay between retries in seconds

Custom Task Classes

Creating Custom Tasks

For complex tasks, create custom task classes that extend BaseTask:

use App\Modules\TaskRunner\BaseTask;

class DatabaseBackupTask extends BaseTask
{
    protected string $database;
    protected string $backupPath;
    protected bool $compress = true;
    protected bool $encrypt = false;

    public function __construct(array $attributes = [])
    {
        parent::__construct($attributes);
        
        // Set default configuration
        $this->setAnalyticsConfiguration([
            'enabled' => true,
            'track_performance' => true,
        ]);
        
        $this->setMonitoringConfiguration([
            'enabled' => true,
            'check_interval' => 30,
        ]);
        
        $this->setRollbackConfiguration([
            'enabled' => true,
            'timeout' => 300,
        ]);
    }

    public function render(): string
    {
        $command = "mysqldump -u {$this->getData('username')} -p{$this->getData('password')} {$this->database}";
        
        if ($this->compress) {
            $command .= " | gzip";
        }
        
        $command .= " > {$this->backupPath}";
        
        if ($this->encrypt) {
            $command .= " && gpg --encrypt --recipient backup-key {$this->backupPath}";
        }
        
        return $command;
    }

    public function getView(): string
    {
        return 'tasks.database-backup';
    }

    public function getViewData(): array
    {
        return [
            'database' => $this->database,
            'backup_path' => $this->backupPath,
            'compress' => $this->compress,
            'encrypt' => $this->encrypt,
            'estimated_size' => $this->estimateBackupSize(),
        ];
    }

    public function validate(): void
    {
        if (empty($this->database)) {
            throw new \InvalidArgumentException('Database name is required');
        }
        
        if (empty($this->backupPath)) {
            throw new \InvalidArgumentException('Backup path is required');
        }
        
        if (!is_writable(dirname($this->backupPath))) {
            throw new \InvalidArgumentException('Backup directory is not writable');
        }
    }

    protected function estimateBackupSize(): string
    {
        // Implementation for estimating backup size
        return '~100MB';
    }

    // Custom methods for this task type
    public function withCompression(bool $compress = true): self
    {
        $this->compress = $compress;
        return $this;
    }

    public function withEncryption(bool $encrypt = true): self
    {
        $this->encrypt = $encrypt;
        return $this;
    }

    public function setDatabase(string $database): self
    {
        $this->database = $database;
        return $this;
    }

    public function setBackupPath(string $path): self
    {
        $this->backupPath = $path;
        return $this;
    }
}

Using Custom Tasks

// Create and configure the task
$backupTask = new DatabaseBackupTask();
$backupTask->setDatabase('myapp');
$backupTask->setBackupPath('/var/backups/myapp.sql');
$backupTask->withCompression(true);
$backupTask->withEncryption(false);

// Set additional data
$backupTask->setData('username', 'root');
$backupTask->setData('password', 'secret');

// Configure callbacks
$backupTask->setCallbackUrl('https://api.example.com/webhooks/backup-complete');
$backupTask->setCallbackConfig([
    'timeout' => 30,
    'retry_attempts' => 3,
]);

// Execute the task
$result = $backupTask->dispatch();

if ($result->successful()) {
    echo "Backup completed successfully!";
    echo "Output: " . $result->output();
} else {
    echo "Backup failed: " . $result->errorOutput();
    echo "Exit code: " . $result->exitCode();
}

Task Factories

Using Task Factories

Task factories provide convenient methods for creating common task types:

use App\Modules\TaskRunner\Examples\DatabaseBackupTask;
use App\Modules\TaskRunner\Examples\DeploymentTask;

// Database backup factory methods
$backupTask = DatabaseBackupTask::forDatabase('myapp', '/var/backups/myapp.sql');
$backupTask = DatabaseBackupTask::create('myapp', '/var/backups/myapp.sql', [
    'compress' => true,
    'encrypt' => true,
]);

// Deployment task factory methods
$deployTask = DeploymentTask::forApp('myapp', '/var/www/myapp', 'git@github.com:user/myapp.git');
$deployTask = DeploymentTask::laravel('myapp', '/var/www/myapp', 'git@github.com:user/myapp.git');
$deployTask = DeploymentTask::nodejs('myapp', '/var/www/myapp', 'git@github.com:user/myapp.git', 'npm');

// Configure the deployment
$deployTask->withPackageManager('composer')
    ->withAssetBuilding('npm run build')
    ->withMigrations('php artisan migrate --force')
    ->withCacheClearing(['php artisan config:clear', 'php artisan route:clear'])
    ->withTesting('php artisan test')
    ->withHealthCheck('https://myapp.com/health')
    ->withSlackNotifications('https://hooks.slack.com/services/...')
    ->withEmailNotifications('admin@example.com');

// Execute the deployment
$result = $deployTask->dispatch();

Creating Custom Factories

class CustomTaskFactory
{
    public static function createBackupTask(string $database, string $path, array $options = []): DatabaseBackupTask
    {
        $task = new DatabaseBackupTask();
        $task->setDatabase($database);
        $task->setBackupPath($path);
        
        if ($options['compress'] ?? false) {
            $task->withCompression(true);
        }
        
        if ($options['encrypt'] ?? false) {
            $task->withEncryption(true);
        }
        
        if (isset($options['username'])) {
            $task->setData('username', $options['username']);
        }
        
        if (isset($options['password'])) {
            $task->setData('password', $options['password']);
        }
        
        return $task;
    }

    public static function createDeploymentTask(string $app, string $path, string $repo, array $options = []): DeploymentTask
    {
        $task = new DeploymentTask();
        $task->setData('app_name', $app);
        $task->setData('deploy_path', $path);
        $task->setData('repo_url', $repo);
        
        if ($options['branch'] ?? false) {
            $task->onBranch($options['branch']);
        }
        
        if ($options['package_manager'] ?? false) {
            $task->withPackageManager($options['package_manager']);
        }
        
        return $task;
    }
}

// Usage
$backupTask = CustomTaskFactory::createBackupTask('myapp', '/var/backups/myapp.sql', [
    'compress' => true,
    'encrypt' => true,
    'username' => 'root',
    'password' => 'secret',
]);

$deployTask = CustomTaskFactory::createDeploymentTask('myapp', '/var/www/myapp', 'git@github.com:user/myapp.git', [
    'branch' => 'main',
    'package_manager' => 'composer',
]);

Task Templates

Using Task Templates

Task templates provide reusable configurations for common task patterns:

use App\Modules\TaskRunner\BaseTask;

class TemplateTask extends BaseTask
{
    public function __construct()
    {
        parent::__construct();
        
        // Configure templates
        $this->setTemplatesConfig([
            'enabled' => true,
            'categories' => [
                'database' => 'Database Operations',
                'deployment' => 'Deployment Tasks',
                'maintenance' => 'System Maintenance',
            ],
        ]);
    }

    public function render(): string
    {
        // Create task from template
        $this->createFromTemplate('database_backup', [
            'database_name' => 'myapp',
            'backup_path' => '/var/backups',
            'compress' => true,
            'encrypt' => false,
        ]);

        return "echo 'Task created from template'";
    }

    public function createProductionTask(): self
    {
        // Create from production template
        return $this->createFromTemplate('production_ready', [
            'monitoring_enabled' => true,
            'analytics_enabled' => true,
            'rollback_enabled' => true,
            'alert_threshold' => 0.8,
        ]);
    }

    public function createDevelopmentTask(): self
    {
        // Create from development template
        return $this->createFromTemplate('development_ready', [
            'monitoring_enabled' => false,
            'analytics_enabled' => true,
            'rollback_enabled' => false,
            'debug_mode' => true,
        ]);
    }
}

Template Management

// Save current task as template
$task->saveAsTemplate('production_deployment', [
    'description' => 'Production deployment with monitoring and rollback',
    'category' => 'production',
    'tags' => ['deployment', 'production', 'monitoring'],
    'version' => '1.0.0',
]);

// List available templates
$templates = $task->listTemplates();

// Search templates
$searchResults = $task->searchTemplates('database');

// Get templates by category
$databaseTemplates = $task->getTemplatesByCategory('database');

// Get popular templates
$popularTemplates = $task->getPopularTemplates(10);

// Get template recommendations
$recommendations = $task->getTemplateRecommendations();

// Validate template parameters
$validation = $task->validateTemplateParameters('database_backup', [
    'database_name' => 'myapp',
    'backup_path' => '/var/backups',
]);

if ($validation['valid']) {
    $task->createFromTemplate('database_backup', $validation['parameters']);
} else {
    echo "Validation errors: " . implode(', ', $validation['errors']);
}

// Export template
$jsonTemplate = $task->exportTemplate('my_template', 'json');

// Import template
$task->importTemplate('/path/to/template.json');

// Clone template
$task->cloneTemplate('source_template', 'new_template_name');

// Get template version history
$versions = $task->getTemplateVersionHistory('my_template');

// Restore template version
$task->restoreTemplateVersion('my_template', '1.2.0');

Task Chains

Creating Task Chains

Task chains allow you to execute multiple tasks in sequence or parallel:

use App\Modules\TaskRunner\TaskChain;
use App\Modules\TaskRunner\AnonymousTask;

// Create a simple chain
$chain = new TaskChain([
    AnonymousTask::command('backup-database', 'mysqldump -u root -p myapp > backup.sql'),
    AnonymousTask::command('compress-backup', 'gzip backup.sql'),
    AnonymousTask::command('upload-backup', 'aws s3 cp backup.sql.gz s3://backups/'),
]);

// Execute the chain
$results = $chain->run();

// Create a chain with options
$chain = new TaskChain([
    AnonymousTask::command('step1', 'echo "Step 1"'),
    AnonymousTask::command('step2', 'echo "Step 2"'),
    AnonymousTask::command('step3', 'echo "Step 3"'),
], [
    'timeout' => 600,
    'stop_on_failure' => true,
    'parallel' => false,
    'max_concurrency' => 1,
]);

// Create a deployment chain
$deploymentChain = new TaskChain([
    AnonymousTask::command('pull-code', 'git pull origin main'),
    AnonymousTask::command('install-deps', 'composer install --no-dev --optimize-autoloader'),
    AnonymousTask::command('run-migrations', 'php artisan migrate --force'),
    AnonymousTask::command('clear-cache', 'php artisan config:clear && php artisan route:clear'),
    AnonymousTask::command('restart-services', 'sudo systemctl restart nginx && sudo systemctl restart php-fpm'),
], [
    'timeout' => 900,
    'stop_on_failure' => true,
    'parallel' => false,
]);

// Execute with progress tracking
$results = $deploymentChain->runWithProgress(function($current, $total, $taskName) {
    echo "Executing {$taskName} ({$current}/{$total})";
});

// Check results
if ($deploymentChain->wasSuccessful()) {
    echo "Deployment completed successfully!";
    echo "Duration: " . $deploymentChain->getDuration() . " seconds";
} else {
    echo "Deployment failed at step: " . $deploymentChain->getFailedStep();
    echo "Error: " . $deploymentChain->getLastError();
}

Advanced Chain Features

// Chain with conditional execution
$chain = new TaskChain([
    AnonymousTask::command('check-disk-space', 'df -h | grep /var'),
    AnonymousTask::command('backup-database', 'mysqldump -u root -p myapp > backup.sql'),
    AnonymousTask::command('verify-backup', 'test -s backup.sql && echo "Backup verified"'),
    AnonymousTask::command('cleanup-old-backups', 'find /var/backups -name "*.sql" -mtime +7 -delete'),
], [
    'stop_on_failure' => true,
    'retry_failed' => true,
    'max_retries' => 3,
]);

// Chain with rollback support
$chain = new TaskChain([
    AnonymousTask::command('backup-current', 'cp -r /var/www/app /var/backups/app-$(date +%Y%m%d)'),
    AnonymousTask::command('deploy-new-version', 'cp -r /tmp/new-version /var/www/app'),
    AnonymousTask::command('run-tests', 'php artisan test'),
    AnonymousTask::command('activate-new-version', 'ln -sf /var/www/app /var/www/current'),
], [
    'rollback_on_failure' => true,
    'rollback_commands' => [
        'restore-backup' => 'cp -r /var/backups/app-$(date +%Y%m%d) /var/www/app',
        'reactivate-old-version' => 'ln -sf /var/www/app /var/www/current',
    ],
]);

// Chain with monitoring
$chain = new TaskChain([
    AnonymousTask::command('pre-deployment-check', 'php artisan health:check'),
    AnonymousTask::command('deploy-application', 'php artisan deploy'),
    AnonymousTask::command('post-deployment-check', 'php artisan health:check'),
], [
    'monitor_health' => true,
    'health_check_url' => 'https://myapp.com/health',
    'health_check_timeout' => 30,
]);

// Execute the chain
$results = $chain->run();

// Get detailed results
foreach ($results as $taskName => $result) {
    echo "Task: {$taskName}\n";
    echo "Status: " . ($result->successful() ? 'Success' : 'Failed') . "\n";
    echo "Duration: " . $result->getDuration() . " seconds\n";
    echo "Output: " . $result->output() . "\n";
    if (!$result->successful()) {
        echo "Error: " . $result->errorOutput() . "\n";
    }
    echo "---\n";
}
