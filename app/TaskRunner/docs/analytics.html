Analytics & Performance Overview

TaskRunner's analytics and performance monitoring system provides comprehensive insights into task execution, resource usage, and optimization opportunities. It helps identify bottlenecks, track performance trends, and optimize task execution for better efficiency.

Performance Metrics

Track execution time, memory usage, CPU utilization, and other key performance indicators.

Resource Monitoring

Monitor system resources including memory, disk I/O, network usage, and database connections.

Optimization Insights

Get intelligent recommendations for improving task performance and resource efficiency.

Key Features

  • • Real-time performance monitoring and metrics collection
  • • Automatic bottleneck detection and performance analysis
  • • Historical trend analysis and performance forecasting
  • • Resource usage optimization recommendations
  • • Cost analysis and efficiency scoring
  • • Performance alerts and threshold monitoring
  • • Comprehensive reporting and data export capabilities

use App\Modules\TaskRunner\BaseTask;

class AnalyticsTask extends BaseTask
{
    public function __construct()
    {
        parent::__construct();
        
        // Enable analytics with detailed configuration
        $this->setAnalyticsConfiguration([
            'enabled' => true,
            'track_performance' => true,
            'track_resources' => true,
            'track_operations' => true,
            'baseline' => [
                'execution_time' => 60, // seconds
                'memory_usage' => 100 * 1024 * 1024, // 100MB
                'cpu_usage' => 0.8, // 80%
            ],
        ]);
    }

    public function render(): string
    {
        // Start performance measurement
        $this->startPerformanceMeasurement('database_operation');
        
        // Simulate database operation
        $script = "echo 'Performing database operation...'";
        
        // End measurement and record metric
        $executionTime = $this->endPerformanceMeasurement('database_operation');
        $this->recordPerformanceMetric('database_execution_time', $executionTime, [
            'operation' => 'database_query',
            'table' => 'users',
        ]);
        
        // Record memory usage
        $memoryUsage = memory_get_peak_usage(true);
        $this->recordPerformanceMetric('memory_peak_usage', $memoryUsage, [
            'unit' => 'bytes',
            'operation' => 'database_operation',
        ]);
        
        return $script;
    }

    public function getPerformanceMetrics(): array
    {
        return [
            'execution_time' => $this->getExecutionTime(),
            'memory_usage' => $this->getMemoryUsage(),
            'cpu_usage' => $this->getCpuUsage(),
            'disk_io' => $this->getDiskIO(),
            'network_usage' => $this->getNetworkUsage(),
        ];
    }

    public function getResourceMetrics(): array
    {
        return [
            'memory' => [
                'peak' => memory_get_peak_usage(true),
                'current' => memory_get_usage(true),
                'limit' => ini_get('memory_limit'),
            ],
            'cpu' => [
                'usage' => $this->getCpuUsage(),
                'load_average' => sys_getloadavg(),
            ],
            'disk' => [
                'read_bytes' => $this->getDiskReadBytes(),
                'write_bytes' => $this->getDiskWriteBytes(),
                'io_operations' => $this->getDiskIOOperations(),
            ],
        ];
    }
}

Resource Monitoring

System Resource Tracking

class ResourceMonitoringTask extends BaseTask
{
    public function __construct()
    {
        parent::__construct();
        
        // Configure resource monitoring
        $this->setAnalyticsConfiguration([
            'enabled' => true,
            'track_resources' => true,
            'resource_thresholds' => [
                'memory_usage' => 0.9, // 90%
                'cpu_usage' => 0.8,    // 80%
                'disk_usage' => 0.85,  // 85%
            ],
        ]);
    }

    public function render(): string
    {
        // Monitor system resources during execution
        $this->startResourceMonitoring();
        
        $script = "echo 'Monitoring system resources...'";
        
        // Record resource metrics at different points
        $this->recordResourceMetric('memory_usage', $this->getMemoryUsage());
        $this->recordResourceMetric('cpu_usage', $this->getCpuUsage());
        $this->recordResourceMetric('disk_usage', $this->getDiskUsage());
        $this->recordResourceMetric('network_io', $this->getNetworkIO());
        
        $this->stopResourceMonitoring();
        
        return $script;
    }

    protected function getMemoryUsage(): float
    {
        $memoryLimit = ini_get('memory_limit');
        $memoryUsed = memory_get_usage(true);
        
        if ($memoryLimit === '-1') {
            return 0.0; // No limit
        }
        
        $memoryLimitBytes = $this->convertToBytes($memoryLimit);
        return $memoryUsed / $memoryLimitBytes;
    }

    protected function getCpuUsage(): float
    {
        // Get CPU usage percentage
        $load = sys_getloadavg();
        $cpuCount = sysconf(_SC_NPROCESSORS_ONLN);
        
        return $load[0] / $cpuCount;
    }

    protected function getDiskUsage(): float
    {
        $path = '/';
        $total = disk_total_space($path);
        $free = disk_free_space($path);
        
        return ($total - $free) / $total;
    }

    protected function getNetworkIO(): array
    {
        // Get network I/O statistics
        $stats = file_get_contents('/proc/net/dev');
        $lines = explode("\n", $stats);
        
        $networkStats = [];
        foreach ($lines as $line) {
            if (preg_match('/^\s*(\w+):\s*(\d+)\s+\d+\s+\d+\s+\d+\s+\d+\s+\d+\s+\d+\s+\d+\s+(\d+)/', $line, $matches)) {
                $interface = $matches[1];
                $bytesReceived = $matches[2];
                $bytesTransmitted = $matches[3];
                
                $networkStats[$interface] = [
                    'bytes_received' => $bytesReceived,
                    'bytes_transmitted' => $bytesTransmitted,
                ];
            }
        }
        
        return $networkStats;
    }

    protected function convertToBytes(string $memoryLimit): int
    {
        $unit = strtolower(substr($memoryLimit, -1));
        $value = (int) substr($memoryLimit, 0, -1);
        
        return match ($unit) {
            'k' => $value * 1024,
            'm' => $value * 1024 * 1024,
            'g' => $value * 1024 * 1024 * 1024,
            default => $value,
        };
    }
}

Resource Thresholds & Alerts

// Set resource thresholds
$task->setAnalyticsConfiguration([
    'resource_thresholds' => [
        'memory_usage' => 0.9,    // Alert at 90% memory usage
        'cpu_usage' => 0.8,       // Alert at 80% CPU usage
        'disk_usage' => 0.85,     // Alert at 85% disk usage
        'disk_io_wait' => 0.1,    // Alert at 10% I/O wait
        'network_errors' => 5,    // Alert after 5 network errors
    ],
    'alert_actions' => [
        'memory_usage' => ['log', 'notify'],
        'cpu_usage' => ['log', 'notify', 'throttle'],
        'disk_usage' => ['log', 'notify', 'cleanup'],
    ],
]);

// Check resource thresholds during execution
if ($this->isResourceThresholdExceeded('memory_usage')) {
    $this->addPerformanceAlert([
        'type' => 'resource_threshold',
        'resource' => 'memory_usage',
        'current_value' => $this->getMemoryUsage(),
        'threshold' => 0.9,
        'severity' => 'warning',
        'message' => 'Memory usage exceeded 90% threshold',
    ]);
}

// Get resource status
$resourceStatus = $this->getResourceStatus();
foreach ($resourceStatus as $resource => $status) {
    echo "{$resource}: " . ($status['healthy'] ? 'Healthy' : 'Warning') . "\n";
    if (!$status['healthy']) {
        echo "  Current: {$status['current']}\n";
        echo "  Threshold: {$status['threshold']}\n";
    }
}

Optimization Insights

Performance Analysis & Recommendations

class OptimizationTask extends BaseTask
{
    public function render(): string
    {
        // Enable optimization insights
        $this->setAnalyticsConfiguration([
            'enabled' => true,
            'optimization_insights' => true,
            'performance_baseline' => [
                'execution_time' => 60,
                'memory_usage' => 100 * 1024 * 1024,
                'cpu_usage' => 0.8,
            ],
        ]);
        
        $script = "echo 'Running optimization analysis...'";
        
        // Get optimization insights
        $insights = $this->getOptimizationInsights();
        
        foreach ($insights as $insight) {
            $this->recordPerformanceMetric('optimization_insight', $insight['score'], [
                'category' => $insight['category'],
                'recommendation' => $insight['recommendation'],
                'impact' => $insight['impact'],
            ]);
        }
        
        return $script;
    }

    public function getOptimizationInsights(): array
    {
        $metrics = $this->getPerformanceMetrics();
        $insights = [];
        
        // Analyze execution time
        if ($metrics['execution_time'] > 60) {
            $insights[] = [
                'category' => 'execution_time',
                'score' => 0.6,
                'recommendation' => 'Consider parallelizing operations or optimizing database queries',
                'impact' => 'high',
                'potential_improvement' => '40-60% reduction in execution time',
            ];
        }
        
        // Analyze memory usage
        if ($metrics['memory_usage'] > 100 * 1024 * 1024) {
            $insights[] = [
                'category' => 'memory_usage',
                'score' => 0.7,
                'recommendation' => 'Implement memory pooling or reduce data loading',
                'impact' => 'medium',
                'potential_improvement' => '30-50% reduction in memory usage',
            ];
        }
        
        // Analyze CPU usage
        if ($metrics['cpu_usage'] > 0.8) {
            $insights[] = [
                'category' => 'cpu_usage',
                'score' => 0.8,
                'recommendation' => 'Consider using background jobs or optimizing algorithms',
                'impact' => 'high',
                'potential_improvement' => '50-70% reduction in CPU usage',
            ];
        }
        
        return $insights;
    }

    public function getEfficiencyScore(): float
    {
        $metrics = $this->getPerformanceMetrics();
        $baseline = $this->getAnalyticsConfiguration()['performance_baseline'];
        
        $executionScore = min(1.0, $baseline['execution_time'] / $metrics['execution_time']);
        $memoryScore = min(1.0, $baseline['memory_usage'] / $metrics['memory_usage']);
        $cpuScore = min(1.0, $baseline['cpu_usage'] / $metrics['cpu_usage']);
        
        // Weighted average
        return ($executionScore * 0.4 + $memoryScore * 0.3 + $cpuScore * 0.3);
    }

    public function getPerformanceTrends(): array
    {
        $historicalMetrics = $this->getHistoricalMetrics();
        
        return [
            'execution_time_trend' => $this->calculateTrend($historicalMetrics['execution_time']),
            'memory_usage_trend' => $this->calculateTrend($historicalMetrics['memory_usage']),
            'cpu_usage_trend' => $this->calculateTrend($historicalMetrics['cpu_usage']),
            'efficiency_trend' => $this->calculateTrend($historicalMetrics['efficiency_score']),
        ];
    }

    protected function calculateTrend(array $values): array
    {
        if (count($values) < 2) {
            return ['direction' => 'stable', 'change_percentage' => 0];
        }
        
        $recent = array_slice($values, -5);
        $older = array_slice($values, -10, 5);
        
        $recentAvg = array_sum($recent) / count($recent);
        $olderAvg = array_sum($older) / count($older);
        
        $changePercentage = (($recentAvg - $olderAvg) / $olderAvg) * 100;
        
        return [
            'direction' => $changePercentage > 0 ? 'improving' : ($changePercentage < 0 ? 'declining' : 'stable'),
            'change_percentage' => abs($changePercentage),
        ];
    }
}
