Events & Broadcasting Overview

TaskRunner's event system provides real-time communication between tasks, chains, and external systems. It enables decoupled architecture with comprehensive event broadcasting and subscription capabilities.

Real-time Events

Instant event broadcasting for task lifecycle and system events.

Event Broadcasting

Broadcast events to multiple channels and subscribers simultaneously.

Event Subscribers

Subscribe to specific events and handle them with custom logic.

Decoupled Architecture

Loose coupling between task execution and event handling.

Event Analytics

Track and analyze event patterns for system optimization.

Secure Events

Secure event transmission with authentication and encryption.

Listening to Events

Event Listeners

Create event listeners to handle specific events:

// Create an event listener
class TaskCompletedListener
{
    public function handle(TaskCompletedEvent $event): void
    {
        // Handle task completion
        Log::info('Task completed', [
            'task_id' => $event->taskId,
            'task_name' => $event->taskName,
            'duration' => $event->duration
        ]);

        // Send notification
        Notification::route('slack', config('notifications.slack.webhook'))
            ->notify(new TaskCompletedNotification($event));

        // Update dashboard
        Cache::put("task_status_{$event->taskId}", 'completed', 3600);
    }
}

// Register the listener
Event::listen(TaskCompletedEvent::class, TaskCompletedListener::class);

Event Subscribers

Create event subscribers for multiple related events:

class TaskEventSubscriber
{
    public function handleTaskStarted(TaskStartedEvent $event): void
    {
        // Handle task start
        Log::info('Task started', ['task_id' => $event->taskId]);
        
        // Update task status
        Task::where('id', $event->taskId)->update([
            'status' => 'running',
            'started_at' => now()
        ]);
    }

    public function handleTaskCompleted(TaskCompletedEvent $event): void
    {
        // Handle task completion
        Log::info('Task completed', ['task_id' => $event->taskId]);
        
        // Update task status
        Task::where('id', $event->taskId)->update([
            'status' => 'completed',
            'completed_at' => now(),
            'duration' => $event->duration
        ]);

        // Send completion notification
        $this->sendCompletionNotification($event);
    }

    public function handleTaskFailed(TaskFailedEvent $event): void
    {
        // Handle task failure
        Log::error('Task failed', [
            'task_id' => $event->taskId,
            'error' => $event->error
        ]);
        
        // Update task status
        Task::where('id', $event->taskId)->update([
            'status' => 'failed',
            'failed_at' => now(),
            'error_message' => $event->error
        ]);

        // Send failure notification
        $this->sendFailureNotification($event);
    }

    public function subscribe(Dispatcher $events): void
    {
        $events->listen(TaskStartedEvent::class, [self::class, 'handleTaskStarted']);
        $events->listen(TaskCompletedEvent::class, [self::class, 'handleTaskCompleted']);
        $events->listen(TaskFailedEvent::class, [self::class, 'handleTaskFailed']);
    }
}

WebSocket Broadcasting

Broadcast events to connected clients:

// Broadcast task events to WebSocket clients
class TaskEventBroadcaster
{
    public function handleTaskCompleted(TaskCompletedEvent $event): void
    {
        broadcast(new TaskCompletedBroadcast($event))->toOthers();
    }

    public function handleTaskProgress(TaskProgressEvent $event): void
    {
        broadcast(new TaskProgressBroadcast($event))->toOthers();
    }
}

// Broadcast event class
class TaskCompletedBroadcast implements ShouldBroadcast
{
    use InteractsWithSockets, SerializesModels;

    public function __construct(public TaskCompletedEvent $event)
    {
    }

    public function broadcastOn(): array
    {
        return ['task-events'];
    }

    public function broadcastAs(): string
    {
        return 'task.completed';
    }

    public function broadcastWith(): array
    {
        return [
            'task_id' => $this->event->taskId,
            'task_name' => $this->event->taskName,
            'status' => 'completed',
            'completed_at' => $this->event->completedAt->toISOString()
        ];
    }
}

Dispatching Events

Automatic Event Dispatching

Events are automatically dispatched by TaskRunner:

class DatabaseBackupTask extends BaseTask
{
    public function handle(): void
    {
        // Task started event is automatically dispatched
        
        // Perform backup
        $this->runCommand('pg_dump production_db > backup.sql');
        
        // Task completed event is automatically dispatched
    }
}

// Events are automatically dispatched:
// - TaskStartedEvent when handle() begins
// - TaskProgressEvent for progress updates
// - TaskCompletedEvent when handle() completes successfully
// - TaskFailedEvent if an exception is thrown

Manual Event Dispatching

Dispatch custom events manually:

class DeploymentTask extends BaseTask
{
    public function handle(): void
    {
        // Dispatch custom deployment event
        event(new DeploymentStartedEvent([
            'version' => 'v2.1.0',
            'environment' => 'production',
            'deployment_id' => $this->getTask()->id
        ]));

        // Perform deployment steps
        $this->runCommand('git pull origin main');
        
        // Dispatch progress event
        event(new DeploymentProgressEvent([
            'step' => 'code_pulled',
            'progress' => 25
        ]));

        $this->runCommand('composer install --no-dev');
        
        event(new DeploymentProgressEvent([
            'step' => 'dependencies_installed',
            'progress' => 50
        ]));

        $this->runCommand('php artisan migrate --force');
        
        // Dispatch completion event
        event(new DeploymentCompletedEvent([
            'version' => 'v2.1.0',
            'environment' => 'production',
            'status' => 'success'
        ]));
    }
}

Event Broadcasting

Broadcast events to external systems:

// Broadcast events to external systems
class ExternalSystemBroadcaster
{
    public function handleTaskCompleted(TaskCompletedEvent $event): void
    {
        // Send to external API
        Http::post('https://api.example.com/webhooks/task-events', [
            'event' => 'task.completed',
            'task_id' => $event->taskId,
            'task_name' => $event->taskName,
            'completed_at' => $event->completedAt->toISOString()
        ]);

        // Send to Slack
        Http::post(config('services.slack.webhook_url'), [
            'text' => "Task {$event->taskName} completed successfully",
            'attachments' => [
                [
                    'fields' => [
                        ['title' => 'Task ID', 'value' => $event->taskId, 'short' => true],
                        ['title' => 'Duration', 'value' => $event->duration . 's', 'short' => true]
                    ]
                ]
            ]
        ]);

        // Send to monitoring system
        Http::post('https://monitoring.example.com/api/events', [
            'type' => 'task_completed',
            'data' => $event->toArray()
        ]);
    }
}

Real-World Examples

Task Monitoring Dashboard

Real-time task monitoring with events:

// Task monitoring subscriber
class TaskMonitoringSubscriber
{
    public function handleTaskStarted(TaskStartedEvent $event): void
    {
        // Update dashboard
        Cache::put("task_{$event->taskId}_status", 'running', 3600);
        Cache::put("task_{$event->taskId}_started", now(), 3600);
        
        // Broadcast to dashboard
        broadcast(new TaskStatusUpdate([
            'task_id' => $event->taskId,
            'status' => 'running',
            'started_at' => now()
        ]))->toOthers();
    }

    public function handleTaskProgress(TaskProgressEvent $event): void
    {
        // Update progress
        Cache::put("task_{$event->taskId}_progress", $event->percentage, 3600);
        
        // Broadcast progress update
        broadcast(new TaskProgressUpdate([
            'task_id' => $event->taskId,
            'percentage' => $event->percentage,
            'message' => $event->message
        ]))->toOthers();
    }

    public function handleTaskCompleted(TaskCompletedEvent $event): void
    {
        // Update final status
        Cache::put("task_{$event->taskId}_status", 'completed', 3600);
        Cache::put("task_{$event->taskId}_completed", now(), 3600);
        Cache::put("task_{$event->taskId}_duration", $event->duration, 3600);
        
        // Broadcast completion
        broadcast(new TaskStatusUpdate([
            'task_id' => $event->taskId,
            'status' => 'completed',
            'completed_at' => now(),
            'duration' => $event->duration
        ]))->toOthers();
        
        // Update analytics
        $this->updateTaskAnalytics($event);
    }

    public function handleTaskFailed(TaskFailedEvent $event): void
    {
        // Update failure status
        Cache::put("task_{$event->taskId}_status", 'failed', 3600);
        Cache::put("task_{$event->taskId}_failed", now(), 3600);
        Cache::put("task_{$event->taskId}_error", $event->error, 3600);
        
        // Broadcast failure
        broadcast(new TaskStatusUpdate([
            'task_id' => $event->taskId,
            'status' => 'failed',
            'failed_at' => now(),
            'error' => $event->error
        ]))->toOthers();
        
        // Send alert
        $this->sendFailureAlert($event);
    }

    private function updateTaskAnalytics(TaskCompletedEvent $event): void
    {
        // Update task performance metrics
        $metrics = [
            'total_tasks' => Cache::increment('total_tasks_completed'),
            'average_duration' => $this->calculateAverageDuration($event->duration),
            'success_rate' => $this->calculateSuccessRate()
        ];
        
        Cache::put('task_analytics', $metrics, 86400);
    }

    private function sendFailureAlert(TaskFailedEvent $event): void
    {
        // Send to monitoring system
        Http::post(config('monitoring.webhook_url'), [
            'alert_type' => 'task_failure',
            'task_id' => $event->taskId,
            'task_name' => $event->taskName,
            'error' => $event->error,
            'failed_at' => now()->toISOString()
        ]);
    }
}

Chain Orchestration

Orchestrate task chains with events:

// Chain orchestration subscriber
class ChainOrchestrationSubscriber
{
    public function handleChainStarted(ChainStartedEvent $event): void
    {
        // Initialize chain tracking
        Cache::put("chain_{$event->chainId}_status", 'running', 3600);
        Cache::put("chain_{$event->chainId}_started", now(), 3600);
        Cache::put("chain_{$event->chainId}_tasks", $event->totalTasks, 3600);
        Cache::put("chain_{$event->chainId}_completed", 0, 3600);
        
        // Notify stakeholders
        $this->notifyChainStarted($event);
    }

    public function handleChainTaskCompleted(ChainTaskCompletedEvent $event): void
    {
        // Update chain progress
        $completedTasks = Cache::increment("chain_{$event->chainId}_completed");
        $totalTasks = Cache::get("chain_{$event->chainId}_tasks");
        $progress = round(($completedTasks / $totalTasks) * 100);
        
        Cache::put("chain_{$event->chainId}_progress", $progress, 3600);
        
        // Broadcast progress
        broadcast(new ChainProgressUpdate([
            'chain_id' => $event->chainId,
            'progress' => $progress,
            'completed_tasks' => $completedTasks,
            'total_tasks' => $totalTasks
        ]))->toOthers();
        
        // Check if chain is complete
        if ($completedTasks >= $totalTasks) {
            event(new ChainCompletedEvent($event->chainId));
        }
    }

    public function handleChainCompleted(ChainCompletedEvent $event): void
    {
        // Update final status
        Cache::put("chain_{$event->chainId}_status", 'completed', 3600);
        Cache::put("chain_{$event->chainId}_completed_at", now(), 3600);
        
        // Broadcast completion
        broadcast(new ChainStatusUpdate([
            'chain_id' => $event->chainId,
            'status' => 'completed',
            'completed_at' => now()
        ]))->toOthers();
        
        // Trigger post-chain actions
        $this->triggerPostChainActions($event->chainId);
    }

    public function handleChainFailed(ChainFailedEvent $event): void
    {
        // Update failure status
        Cache::put("chain_{$event->chainId}_status", 'failed', 3600);
        Cache::put("chain_{$event->chainId}_failed_at", now(), 3600);
        Cache::put("chain_{$event->chainId}_error", $event->error, 3600);
        
        // Broadcast failure
        broadcast(new ChainStatusUpdate([
            'chain_id' => $event->chainId,
            'status' => 'failed',
            'failed_at' => now(),
            'error' => $event->error
        ]))->toOthers();
        
        // Trigger rollback if configured
        $this->triggerChainRollback($event->chainId);
    }

    private function triggerPostChainActions(string $chainId): void
    {
        // Send completion notifications
        $this->sendChainCompletionNotifications($chainId);
        
        // Update analytics
        $this->updateChainAnalytics($chainId);
        
        // Trigger dependent processes
        $this->triggerDependentProcesses($chainId);
    }

    private function triggerChainRollback(string $chainId): void
    {
        // Check if rollback is configured
        $rollbackConfig = Cache::get("chain_{$chainId}_rollback_config");
        
        if ($rollbackConfig && $rollbackConfig['enabled']) {
            event(new ChainRollbackEvent($chainId, $rollbackConfig));
        }
    }
}
