BackgroundTaskTracker

Manages the complete lifecycle of background task tracking with real-time monitoring, callback handling, and status management. 

use App\Modules\TaskRunner\Services\BackgroundTaskTracker;
use App\Modules\TaskRunner\Models\Task;

$tracker = app(BackgroundTaskTracker::class);

// Start tracking a task
$tracker->startTracking($taskModel, $taskInstance);

// Monitor task progress
$tracker->monitorTask($taskModel, $taskInstance);

// Handle task completion
$tracker->handleTaskCompletion($taskModel, $taskInstance);

// Handle task failure
$tracker->handleTaskFailure($taskModel, $taskInstance, $error);

// Handle task timeout
$tracker->handleTaskTimeout($taskModel, $taskInstance);

// Get running tasks
$runningTasks = $tracker->getRunningTasks();

// Get task status
$status = $tracker->getTaskStatus($taskModel);

// Cancel tracking
$tracker->cancelTracking($taskModel, $taskInstance);

// Clean up old tasks
$deletedCount = $tracker->cleanupOldTasks(30);

CallbackService

Handles HTTP callback delivery with retry logic, error handling, and comprehensive logging. 

use App\Modules\TaskRunner\Services\CallbackService;
use App\Modules\TaskRunner\Enums\CallbackType;

$callbackService = app(CallbackService::class);

// Send a callback
$success = $callbackService->send($task, CallbackType::Finished, [
    'event' => 'task_completed',
    'success' => true,
    'duration' => 120
]);

// Send with custom configuration
$success = $callbackService->sendWithConfig(
    'https://api.example.com/webhook',
    ['task_id' => '123', 'status' => 'completed'],
    ['Authorization' => 'Bearer token'],
    30,
    ['max_attempts' => 5, 'delay' => 10]
);

// Send batch callbacks
$callbacks = [
    ['task' => $task1, 'type' => CallbackType::Started, 'data' => []],
    ['task' => $task2, 'type' => CallbackType::Finished, 'data' => []],
];
$results = $callbackService->sendBatch($callbacks);

// Test callback URL
$isReachable = $callbackService->testCallbackUrl('https://api.example.com/webhook');

// Validate callback data
$isValid = $callbackService->validateCallbackData([
    'task_id' => '123',
    'callback_type' => 'finished',
    'timestamp' => now()->toISOString()
]);

MonitoringService

Provides comprehensive monitoring capabilities including health checks, performance metrics, and alerting. 

use App\Modules\TaskRunner\Services\MonitoringService;

$monitoring = app(MonitoringService::class);

// Start monitoring a task
$monitoring->startMonitoring($taskId, [
    'health_check_interval' => 30,
    'alert_thresholds' => [
        'memory_usage' => 80,
        'cpu_usage' => 90,
        'execution_time' => 3600
    ]
]);

// Check task health
$health = $monitoring->checkTaskHealth($taskId);

if ($health->isHealthy()) {
    echo "Task is healthy\n";
} else {
    echo "Task health issues: " . implode(', ', $health->getIssues()) . "\n";
}

// Get performance metrics
$metrics = $monitoring->getTaskMetrics($taskId);
echo "Memory usage: {$metrics['memory_usage']}%\n";
echo "CPU usage: {$metrics['cpu_usage']}%\n";
echo "Execution time: {$metrics['execution_time']} seconds\n";

// Set up alerts
$monitoring->setAlert($taskId, 'memory_usage', 80, function ($taskId, $value) {
    Log::warning("Task {$taskId} memory usage high: {$value}%");
});

// Get monitoring dashboard data
$dashboard = $monitoring->getDashboardData();

// Stop monitoring
$monitoring->stopMonitoring($taskId);

RollbackService

Manages task rollback operations with checkpoint creation, state restoration, and recovery procedures. 

use App\Modules\TaskRunner\Services\RollbackService;

$rollback = app(RollbackService::class);

// Create a rollback checkpoint
$checkpointId = $rollback->createCheckpoint($taskId, [
    'description' => 'Before database migration',
    'data' => [
        'database_version' => '1.2.3',
        'backup_file' => '/backups/pre_migration.sql',
        'config_snapshot' => '/config/backup.json'
    ]
]);

// Execute risky operation
try {
    $task->runCommand('php artisan migrate --force');
    $rollback->commitCheckpoint($checkpointId);
} catch (Exception $e) {
    // Rollback on failure
    $rollback->rollbackToCheckpoint($checkpointId);
    throw $e;
}

// Create multiple checkpoints
$checkpoint1 = $rollback->createCheckpoint($taskId, 'backup-created');
$checkpoint2 = $rollback->createCheckpoint($taskId, 'code-updated');
$checkpoint3 = $rollback->createCheckpoint($taskId, 'migrations-run');

// Rollback to specific checkpoint
$rollback->rollbackToCheckpoint($checkpoint2);

// Get rollback history
$history = $rollback->getRollbackHistory($taskId);
foreach ($history as $entry) {
    echo "Checkpoint: {$entry['checkpoint_id']}\n";
    echo "Description: {$entry['description']}\n";
    echo "Created: {$entry['created_at']}\n";
    echo "Status: {$entry['status']}\n";
}

// Clean up old checkpoints
$rollback->cleanupOldCheckpoints($taskId, 30); // Keep last 30 days

    'category' => 'deployment',

        'tasks' => [
            [
                'name' => 'backup-database',
                'command' => 'php artisan backup:run',
                'timeout' => 300
            ],
            [
                'name' => 'pull-code',

ConditionalStreamingService

Provides intelligent streaming with conditional output filtering, rate limiting, and content transformation. 

use App\Modules\TaskRunner\Services\ConditionalStreamingService;

$streaming = app(ConditionalStreamingService::class);

// Configure streaming conditions
$streaming->setConditions([
    'include_debug' => false,
    'min_log_level' => 'info',
    'max_output_length' => 1000,
    'rate_limit' => 100 // messages per second
]);

// Add output filters
$streaming->addFilter(function ($message, $level) {
    // Filter out sensitive information
    return str_replace(['password', 'secret'], '***', $message);
});

$streaming->addFilter(function ($message, $level) {
    // Only include error and warning messages
    return in_array($level, ['error', 'warning']) ? $message : null;
});

// Stream with conditions
$streaming->stream('Task started', 'info');
$streaming->stream('Processing data...', 'debug'); // Will be filtered out
$streaming->stream('Error: Connection failed', 'error'); // Will be included

// Get filtered output
$filteredOutput = $streaming->getFilteredOutput();

// Clear streaming buffer
$streaming->clearBuffer();

// Set streaming mode
$streaming->setMode('real-time'); // or 'buffered', 'conditional'
