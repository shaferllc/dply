Core Components

Complete reference for all TaskRunner core components and their capabilities.

BaseTask

The foundation class for all tasks in TaskRunner. Provides core functionality for task execution, output handling, and callback support. 

use App\Modules\TaskRunner\BaseTask;
use App\Modules\TaskRunner\Enums\CallbackType;

class MyCustomTask extends BaseTask
{
    public function handle(): void
    {
        // Configure the task
        $this->setName('my-custom-task')
            ->setTimeout(300) // 5 minutes
            ->setCallbackUrl('https://api.example.com/webhooks/task')
            ->enableCallbacks();

        // Send started callback
        $this->sendCallback(CallbackType::Started, [
            'event' => 'task_started',
            'custom_data' => 'value'
        ]);

        // Execute your task logic
        $this->runCommand('echo "Hello World"');
        
        // Send progress updates
        $this->sendProgressCallback([
            'progress' => 50,
            'message' => 'Halfway done'
        ]);

        // Send completion callback
        $this->sendCallback(CallbackType::Finished, [
            'event' => 'task_completed',
            'success' => true
        ]);
    }
}

Key Methods

Configuration

  • setName(string $name) - Set task name
  • setTimeout(int $seconds) - Set execution timeout
  • setCallbackUrl(string $url) - Set callback URL
  • enableCallbacks() - Enable callback support
  • setCallbackConfig(array $config) - Configure callbacks

Execution

  • runCommand(string $command) - Execute shell command
  • runScript(string $script) - Execute script file
  • getOutput() - Get task output
  • getExitCode() - Get exit code
  • isSuccessful() - Check if task succeeded

EnhancedTask

Extended task class with advanced features including streaming, analytics, monitoring, and rollback capabilities. 

use App\Modules\TaskRunner\EnhancedTask;
use App\Modules\TaskRunner\Enums\CallbackType;

class AdvancedTask extends EnhancedTask
{
    public function handle(): void
    {
        // Enable advanced features
        $this->enableStreaming()
            ->enableAnalytics()
            ->enableMonitoring()
            ->enableRollback();

        // Configure rollback
        $this->setRollbackConfig([
            'enabled' => true,
            'checkpoint_interval' => 30,
            'max_checkpoints' => 10
        ]);

        // Create checkpoint before risky operation
        $this->createCheckpoint('before-database-update');

        // Execute risky operation
        $this->runCommand('php artisan migrate --force');

        // If successful, commit checkpoint
        $this->commitCheckpoint('before-database-update');

        // Send analytics data
        $this->recordMetric('database_migration_time', microtime(true));
        $this->recordMetric('migrations_applied', 5);

        // Stream real-time updates
        $this->stream('Database migration completed successfully');
    }

    protected function rollback(): void
    {
        // Implement rollback logic
        $this->runCommand('php artisan migrate:rollback --step=1');
        $this->stream('Rollback completed');
    }
}

ProcessRunner

Core process execution engine with retry logic, streaming output, and comprehensive error handling. 

use App\Modules\TaskRunner\ProcessRunner;
use Illuminate\Process\PendingProcess;

$processRunner = app(ProcessRunner::class);

// Execute a process with output handling
$output = $processRunner->run(
    Process::command('php artisan migrate --force')
        ->timeout(300)
        ->tty(),
    
    // Output callback
    function (string $type, string $buffer) {
        echo "[{$type}] {$buffer}";
    }
);

// Check results
if ($output->isSuccessful()) {
    echo "Process completed successfully\n";
    echo "Output: " . $output->getBuffer() . "\n";
} else {
    echo "Process failed with exit code: " . $output->getExitCode() . "\n";
    echo "Error output: " . $output->getErrorOutput() . "\n";
}

Features

Retry Logic

  • Automatic retry on failure (max 3 attempts)
  • Exponential backoff between retries
  • Configurable retry conditions
  • Timeout handling with retry

Output Handling

  • Real-time output streaming
  • Separate stdout/stderr handling
  • Output buffering and retrieval
  • Custom output callbacks

TaskDispatcher

Main dispatcher for executing tasks with connection management, output handling, and result processing. 

use App\Modules\TaskRunner\TaskDispatcher;
use App\Modules\TaskRunner\PendingTask;

$dispatcher = app(TaskDispatcher::class);

// Create a pending task
$pendingTask = PendingTask::make()
    ->task(new MyCustomTask())
    ->onConnection($connection)
    ->withTimeout(300)
    ->withOptions(['verbose' => true]);

// Execute the task
$result = $dispatcher->run($pendingTask);

// Handle the result
if ($result->isSuccessful()) {
    echo "Task completed successfully\n";
    echo "Output: " . $result->getOutput() . "\n";
    echo "Duration: " . $result->getDuration() . " seconds\n";
} else {
    echo "Task failed: " . $result->getError() . "\n";
}

// Execute with real-time output
$dispatcher->run($pendingTask, function (string $type, string $buffer) {
    echo "[{$type}] {$buffer}";
});

Key Features

Execution

  • runWithOutput(PendingTask $task, callable $callback) - Execute with output callback
  • dispatch(PendingTask $task) - Dispatch to queue
  • dispatchSync(PendingTask $task) - Execute synchronously

Management

  • Connection management
  • Timeout handling
  • Error processing
  • Result formatting

TaskChain

Execute multiple tasks in sequence with dependency management, error handling, and rollback support. 

use App\Modules\TaskRunner\TaskChain;
use App\Modules\TaskRunner\PendingTask;

$chain = new TaskChain();

// Add tasks to the chain
$chain->addTask(
    PendingTask::make()
        ->task(new BackupDatabaseTask())
        ->withName('backup-database')
        ->withTimeout(600)
);

$chain->addTask(
    PendingTask::make()
        ->task(new UpdateCodeTask())
        ->withName('update-code')
        ->withTimeout(300)
        ->dependsOn('backup-database') // Only run after backup
);

$chain->addTask(
    PendingTask::make()
        ->task(new RunMigrationsTask())
        ->withName('run-migrations')
        ->withTimeout(180)
        ->dependsOn('update-code')
);

$chain->addTask(
    PendingTask::make()
        ->task(new ClearCacheTask())
        ->withName('clear-cache')
        ->withTimeout(60)
        ->dependsOn('run-migrations')
);

// Execute the chain
$results = $chain->execute();

// Handle results
foreach ($results as $taskName => $result) {
    if ($result->isSuccessful()) {
        echo "✅ {$taskName}: Completed successfully\n";
    } else {
        echo "❌ {$taskName}: Failed - " . $result->getError() . "\n";
    }
}

// Execute with rollback support
$chain->executeWithRollback();

ParallelTaskExecutor

Execute multiple tasks in parallel with concurrency control, resource management, and result aggregation. 

use App\Modules\TaskRunner\ParallelTaskExecutor;
use App\Modules\TaskRunner\PendingTask;

$executor = new ParallelTaskExecutor();

// Configure concurrency
$executor->setMaxConcurrency(5)
    ->setTimeout(1800) // 30 minutes
    ->enableProgressTracking();

// Add tasks for parallel execution
$executor->addTask(
    PendingTask::make()
        ->task(new ProcessServer1Task())
        ->withName('server-1-processing')
);

$executor->addTask(
    PendingTask::make()
        ->task(new ProcessServer2Task())
        ->withName('server-2-processing')
);

$executor->addTask(
    PendingTask::make()
        ->task(new ProcessServer3Task())
        ->withName('server-3-processing')
);

// Execute tasks in parallel
$results = $executor->execute();

// Handle aggregated results
$successCount = 0;
$failureCount = 0;

foreach ($results as $taskName => $result) {
    if ($result->isSuccessful()) {
        $successCount++;
        echo "✅ {$taskName}: Completed\n";
    } else {
        $failureCount++;
        echo "❌ {$taskName}: Failed - " . $result->getError() . "\n";
    }
}

echo "Summary: {$successCount} successful, {$failureCount} failed\n";

// Execute with progress callback
$executor->execute(function (int $completed, int $total) {
    $percentage = round(($completed / $total) * 100);
    echo "Progress: {$completed}/{$total} ({$percentage}%)\n";
});

Connection Management

Manage connections to remote servers with authentication, SSH key management, and connection pooling. 

use App\Modules\TaskRunner\Connection;
use App\Modules\TaskRunner\ConnectionManager;

$manager = app(ConnectionManager::class);

// Create a connection
$connection = new Connection([
    'host' => '192.168.1.100',
    'port' => 22,
    'username' => 'deploy',
    'private_key' => '/path/to/private/key',
    'timeout' => 30,
    'retry_attempts' => 3
]);

// Test the connection
if ($connection->test()) {
    echo "Connection successful\n";
} else {
    echo "Connection failed: " . $connection->getLastError() . "\n";
}

// Use connection with task
$pendingTask = PendingTask::make()
    ->task(new MyTask())
    ->onConnection($connection);

// Get connection as different user
$rootConnection = $connection->asUser('root');
$sudoConnection = $connection->withSudo();

// Connection pooling
$pooledConnection = $manager->getConnection('production-server-1');
$manager->releaseConnection($pooledConnection);
